# 给左边 div 设置 float:left，给右边 div 设置 margin-left

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <title>Document</title>
    <style>
      .box {
        overflow: hidden;
        border: 1px solid #000;
      }
      .left {
        float: left;
        width: 200px;
        height: 200px;
        background-color: green;
      }
      .right {
        margin-left: 200px;
        height: 150px;
        background-color: yellow;
      }
    </style>
  </head>
  <body>
    <div class="box">
      <div class="left">左边固定</div>
      <div class="right">右边自适应</div>
    </div>
  </body>
</html>

# 左边 div 绝对定位，右边 div 设置 margin-left

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <title>Document</title>
    <style>
      .box {
        position: relative;
        height: 200px;
        border: 1px solid #000;
      }
      .left {
        position: absolute;
        left: 0;
        top: 0;
        width: 200px;
        height: 100%;
        background-color: green;
      }
      .right {
        margin-left: 200px;
        height: 150px;
        background-color: yellow;
      }
    </style>
  </head>
  <body>
    <div class="box">
      <div class="left">左边固定</div>
      <div class="right">右边自适应</div>
    </div>
  </body>
</html>

# 给父盒子设置 display:table，给子盒子设置 display:table-cell

<html lang="en">
  <head>
    <meta charset="UTF-8" />

        display: table;

    <div class="box">
      <div class="left">左边固定</div>

</html>

# 双 float + calc()

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <title>Document</title>
    <style>
      .box {
        overflow: hidden;

      }

    <div class="box">
      <div class="left">左边固定</div>

  </body>

# 双 inline-block + calc()

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <title>Document</title>
    <style>
      .box {
        box-sizing: content-box;
        border: 1px solid #000;
        font-size: 0;
      }
      .left,
      .right {
        display: inline-block;
        vertical-align: top;
        box-sizing: border-box;
        width: 200px;
        height: 200px;
        background-color: green;
      }
      .right {
        width: calc(100% - 200px);
        height: 150px;
        background-color: yellow;
      }
    </style>
  </head>
  <body>
    <div class="box">
      <div class="left">左边固定</div>
      <div class="right">右边自适应</div>
    </div>
  </body>
</html>

# grid 布局

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>测试代码</title>

    <style>
      .box {
        display: grid;
        height: 400px;
        grid-template-columns: 200px 1fr;
      }
      .left {
        background-color: red;
      }

      .right {
        background-color: green;
      }
    </style>
  </head>
  <body>
    <div class="box">
      <div class="left">左边固定</div>
      <div class="right">右边自适应</div>
    </div>
  </body>
</html>
